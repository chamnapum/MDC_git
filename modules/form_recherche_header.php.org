<script type="text/javascript">
	function changeAction(page) {
		//alert(page);
		//document.rech.action = page;
		$('#formHaut').attr('action',page);

	$(document).ready(function() {
		
	function myAutoComplete(de){
		var de = $('#departement').val();
		var ville = '<?php echo $Value_ville;?>';
		if(de=='near'){
			$('.ville_nearby').html('<input type="text" style="width:175px; margin:0px !important; " disabled="disabled">');
		}else if(de=='region'){
			$('.ville_nearby').html('<input type="text" style="width:175px; margin:0px !important; " disabled="disabled">');
		}else{
			var dataString = 'de='+de+'&ville='+ville;
			$.ajax({
					type: "POST",
					url: "ajax/ville_near_multi.php",
					data: dataString,
					cache: false,
					success: function(datas){
						$(".ville_nearby").html(datas);
					}
				});	
			return false;
		}
	}
		
		var de = $('#departement').val();
		myAutoComplete(de);

			var val = $('input:radio[name=choix]:checked').val();
			var dataString = 'val='+val;
			$.ajax({
					type: "POST",
					url: "ajax/category2.php",
					data: dataString,
					cache: false,
					success: function(datas){
						$("#cate_search").html(datas);
					}
				});	
			return false;
		});	
		
		cate_search();		
		function cate_search(){
			
			var val = $('input:radio[name=choix]:checked').val();
			
			if(val==null){
				val = 3;
			}
			
			if(val == 0){
				changeAction('produit.html');
			}else if(val == 1){
				changeAction('coupon.html');
			}else if(val == 3){
				changeAction('magasin.html');
			}
			var dataString = 'val='+val+'&categorie=<?php echo $_REQUEST['categorie'];?>'+'&sous_categorie=<?php echo $_REQUEST['sous_categorie'];?>';
			$.ajax({
					type: "POST",
					url: "ajax/category2.php",
					data: dataString,
					cache: false,
					success: function(datas){
						$("#cate_search").html(datas);
					}
				});	
			return false;
		}
		
		$('#departement').change(function() {
			de = $('#departement').val();
			var ville = '<?php echo $Value_ville;?>';
			
			$("#ville").val('');
			$("#id_ville").val('');
			
			var de_txt = $("#departement option:selected").text();
			$("#departement_text").val(de_txt);

			if(de=='near'){
				$('.ville_nearby').html('<input type="text" style="width:175px; margin:0px !important; " disabled="disabled">');
			}else if(de=='region'){
				$('.ville_nearby').html('<input type="text" style="width:175px; margin:0px !important; " disabled="disabled">');
			}else{
				
				var dataString = 'de='+de+'&ville='+ville;
				$.ajax({
						type: "POST",
						url: "ajax/ville_near_multi.php",
						data: dataString,
						cache: false,
						success: function(datas){
							$(".ville_nearby").html(datas);
						}
					});	
				return false;
			}
		});	

		$('#cate').change(function() {
			var css = $(this).find('option:selected').attr("class");
			var value = $('#cate').val();
			if(css=='categorie'){
				$('#categorie').val(value);
				$('#sous_categorie').val('');
			}else if(css=='sous_categorie'){
				$('#sous_categorie').val(value);
				$('#categorie').val('');
			}
		});
		
		
		
	});
</script>
